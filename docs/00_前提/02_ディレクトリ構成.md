│  │  ├─ featurea_list.tsx   # 一覧UIコンポーネント（機能A）
│  │  ├─ featurea_details.tsx# 詳細UIコンポーネント（機能A）
│  │  ├─ featurea_slice.tsx  # 状態管理・ビジネスロジック（機能A）
│  │  └─ featurea_api.tsx    # API通信ロジック（機能A）
│  ├─ featureb                 # 機能B関連
│  │  ├─ protos              # 機能B専用 grpc/quic protoファイル
│  │  ├─ featureb_list.tsx   # 一覧UIコンポーネント（機能B）
│  │  ├─ featureb_details.tsx# 詳細UIコンポーネント（機能B）
│  │  ├─ featureb_slice.tsx  # 状態管理・ビジネスロジック（機能B）
│  │  └─ featureb_api.tsx    # API通信ロジック（機能B）
│  └─ featurec                 # 機能C関連
│     ├─ protos              # 機能C専用 grpc/quic protoファイル
│     ├─ featurec_list.tsx   # 一覧UIコンポーネント（機能C）
│     ├─ featurec_details.tsx# 詳細UIコンポーネント（機能C）
│     ├─ featurec_slice.tsx  # 状態管理・ビジネスロジック（機能C）
│     └─ featurec_api.tsx    # API通信ロジック（機能C）
├─ shared                     # 複数機能で共有する共通リソース
│  ├─ protos                # 共通 grpc/quic protoファイル
│  ├─ components           # 共通UIコンポーネント
│  │  ├─ header.tsx         # 共通ヘッダー（レイアウト用）
│  │  ├─ footer.tsx         # 共通フッター（レイアウト用）
│  │  ├─ menubar.tsx        # 共通メニューバー
│  │  ├─ button.tsx         # 共通ボタンコンポーネント
│  │  ├─ textbox.tsx        # 共通テキストボックス
│  │  ├─ modal.tsx          # 共通モーダルウィンドウ
│  │  ├─ debugtoolbar.tsx   # デバッグ用ツールバー
│  │  ├─ searchbar.tsx      # 検索バーコンポーネント
│  │  └─ notification.tsx   # 通知システムコンポーネント
│  ├─ authcontext.tsx       # 認証情報管理コンテキスト
│  ├─ logger.tsx            # ログ出力ユーティリティ
│  ├─ logic                # 共通ロジック群
│  │  ├─ authlogic.tsx      # 認証関連ロジック
│  │  ├─ dataprocessing.tsx # データ処理ロジック
│  │  ├─ validation.tsx     # 入力検証ロジック
│  │  ├─ mockapilogic.tsx   # APIモック処理ロジック
│  │  └─ errorhandlinglogic.tsx # エラーハンドリングロジック
│  ├─ state                # 状態管理
│  │  ├─ globalstate.tsx    # アプリ全体の状態管理
│  │  ├─ userstate.tsx      # ユーザー情報状態管理
│  │  ├─ productstate.tsx   # 商品情報状態管理

│  │  ├─ dateutils.tsx      # 日付操作ユーティリティ
│  │  ├─ errorhandler.tsx   # エラーハンドリングユーティリティ
│  │  ├─ apiclient.tsx      # API通信ユーティリティ
│  │  ├─ grpcclient.tsx     # gRPC通信ユーティリティ
│  │  ├─ quicclient.tsx     # QUIC通信ユーティリティ
│  │  └─ themeutils.tsx     # テーマ切替ユーティリティ
│  ├─ hooks                # カスタム React フック
│  │  ├─ usefetch.tsx       # データ取得用フック
│  │  ├─ usedebounce.tsx    # デバウンス処理用フック
│  │  ├─ usetheme.tsx       # テーマ変更用フック
│  │  └─ usenotification.tsx# 通知用フック
│  ├─ types                # 共通型定義

│  ├─ test                 # テスト関連ファイル
│  │  ├─ testutils.tsx      # テスト補助ユーティリティ
│  │  └─ coveragereport.tsx # テストカバレッジレポート生成
│  ├─ theme.tsx             # テーマ設定
│  ├─ styles               # スタイルシート
│  │  ├─ colors.css         # カラーパレット設定
│  │  └─ typography.css     # タイポグラフィ設定
│  └─ websocket            # WebSocket関連機能
│     ├─ websocket_client.tsx  # WebSocketクライアントロジック
│     └─ websocket_context.tsx # WebSocketコンテキスト管理
