# ディレクトリ構成

```
react
├── client
│   └── sample1
│       ├── config
│       │   ├── clientSettings.yaml         // クライアント側設定ファイル（include設定）
│       │   ├── uiSettings.yaml             // UI関連設定
│       │   ├── networkSettings.yaml        // ネットワーク関連設定
│       │   ├── themeSettings.yaml          // テーマ設定ファイル
│       │   └── featureFlags.yaml           // フィーチャーフラグ設定
│       └── src
│           ├── feature
│           │   └── Login
│           │       ├── LoginPage.tsx        // ログインページ
│           │       └── LoginHelper.ts       // ログイン補助ファイル
│           ├── shared
│           │   ├── layout
│           │   │   ├── Header.tsx           // ヘッダー
│           │   │   ├── Footer.tsx           // フッター
│           │   │   ├── Body.tsx             // ボディ
│           │   │   └── BasePage.tsx         // ベースページ（ヘッダー、ボディ、フッターを内包）
│           │   ├── components
│           │   │   ├── Button.tsx           // ボタンコンポーネント
│           │   │   ├── TextField.tsx        // テキストフィールドコンポーネント
│           │   │   ├── Modal.tsx            // モーダルコンポーネント
│           │   │   └── Alert.tsx            // アラートコンポーネント
│           │   ├── error
│           │   │   └── ErrorBoundary.tsx    // エラーハンドリング
│           │   ├── network
│           │   │   └── APIHandler.ts        // 通信処理（REST API, gRPC, QUIC, WebSocket の実装）
│           │   ├── routing
│           │   │   ├── AppRoutes.tsx        // アプリ全体のルート定義
│           │   │   ├── PrivateRoute.tsx     // 認証が必要なルート
│           │   │   └── RouteConfig.tsx      // ルート設定の管理
│           │   └── state
│           │       ├── store.ts             // Redux ストア設定
│           │       └── slices
│           │           ├── userSlice.ts     // ユーザー関連の状態管理
│           │           └── productSlice.ts  // 商品関連の状態管理
│           └── asset
│               ├── images
│               │   └── logo.png             // 画像ファイル
│               ├── styles
│               │   └── main.css             // スタイルシート
│               └── fonts
│                   └── custom.ttf           // フォントファイル
└── server
    └── sample1
        ├── config
        │   ├── serverSettings.yaml          // サーバー側設定ファイル（include設定）
        │   ├── database.yaml                // データベース接続・クエリ設定
        │   ├── logging.yaml                 // ロギング設定
        │   └── security.yaml                // 認証・認可、CSRF対策等の設定
        └── src
            ├── feature
            │   └── LoginAPI
            │       ├── Controller.cs      // APIコントローラー
            │       ├── Service.cs         // サービス層
            │       └── Repository.cs      // リポジトリ層
            └── shared
                ├── database
                │   └── DatabaseHandler.cs // データベース接続・クエリ処理
                ├── logging
                │   └── Logger.cs          // ロギング処理
                ├── config
                │   └── serverConfig.yaml  // 設定ファイル読込処理用
                ├── network
                │   └── NetworkService.cs  // 通信処理（REST API, gRPC, QUIC, WebSocket の実装）
                ├── security
                │   └── SecurityManager.cs // 認証、認可、CSRF対策等
                └── cache
                    └── CacheService.cs    // キャッシュ処理（Redis、Memcached 等）
```
